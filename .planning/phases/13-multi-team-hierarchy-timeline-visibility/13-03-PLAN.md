  <action>Create new store file:
  
  ```typescript
  import { writable } from 'svelte/store';
  
  interface SubTeam {
    id: number;
    name: string;
    supervisor_id: number | null;
  }
  
  const { subscribe, set, update } = writable<SubTeam | null>(null);
  
  // Load from localStorage on init
  if (typeof window !== 'undefined') {
    const stored = localStorage.getItem('selectedSubTeam');
    if (stored) set(JSON.parse(stored));
  }
  
  // Persist to localStorage on change
  subscribe((value) => {
    if (typeof window !== 'undefined') {
      if (value) {
        localStorage.setItem('selectedSubTeam', JSON.stringify(value));
      } else {
        localStorage.removeItem('selectedSubTeam');
      }
    }
  });
  
  export const subTeamStore = { subscribe, set, update };
  ```
  
  Per RESEARCH.md Pattern 3: localStorage for switcher state persistence.
  Per Claude's Discretion: Using localStorage (simpler than URL param or backend session).

<task type="auto">
  <name>Task 3: Add global sub-team switcher to layout sidebar</name>
  <files>frontend/src/routes/+layout.svelte</files>
  <action>Add SubTeamSwitcher component to sidebar (below nav items, above user section, per D-09):
  
  ```svelte
  <script>
    import { subTeamStore } from '$lib/stores/subTeam';
    import { onMount } from 'svelte';
    import { ChevronDown } from 'lucide-svelte';
    import { api } from '$lib/api';
    import { authStore } from '$lib/stores/auth';
    
    let subTeams = [];
    let expanded = false;
    let selectedSubTeam = null;
    
    $: isAdmin = $authStore?.user?.role === 'admin';
    
    onMount(async () => {
      if (isAdmin) {
        subTeams = await api.sub_teams.list();
      }
    });
    
    subTeamStore.subscribe(v => selectedSubTeam = v);
    
    function selectSubTeam(subTeam) {
      subTeamStore.set(subTeam);
      expanded = false;
      window.location.reload(); // Per D-10: all pages auto-refresh with filtered data
    }
    
    function selectAllTeams() {
      subTeamStore.set(null);
      expanded = false;
      window.location.reload();
    }
  </script>
  
  {#if isAdmin}
  <div class="sub-team-switcher">
    <button on:click={() => expanded = !expanded} class="switcher-button">
      {selectedSubTeam ? selectedSubTeam.name : 'All Teams'}
      <ChevronDown />
    </button>
    {#if expanded}
    <div class="dropdown">
      <button on:click={selectAllTeams} class="dropdown-item">All Teams</button>
      {#each subTeams as subTeam}
      <button 
        on:click={() => selectSubTeam(subTeam)} 
        class:dropdown-item-selected={selectedSubTeam?.id === subTeam.id}
        class="dropdown-item"
      >
        {subTeam.name}
      </button>
      {/each}
    </div>
    {/if}
  </div>
  {/if}
  
  <style>
  .sub-team-switcher {
    @apply px-3 py-2 bg-gray-800/50 border border-gray-700 rounded-lg;
  }

  }
  .dropdown-item-selected {
    @apply bg-primary-500/20 text-primary-400;
  }
  </style>
  ```
