This component:
- Receives props: `viewMode` (string: 'project' | 'member'), `rangeType` (string: 'week' | 'month' | 'custom'), `rangeStart` (Date), `rangeEnd` (Date)
- Emits events: `viewchange` (new viewMode), `rangechange` ({type, start, end})

```svelte
<script lang="ts">
    import { createEventDispatcher } from 'svelte';
    import { GanttChartSquare, Users } from 'lucide-svelte';

    export let viewMode: 'project' | 'member' = 'project';
    export let rangeType: 'week' | 'month' | 'custom' = 'month';
    export let rangeStart: Date = new Date();
    export let rangeEnd: Date = new Date();

    const dispatch = createEventDispatcher();

    function setViewMode(mode: 'project' | 'member') {
        viewMode = mode;
        dispatch('viewchange', mode);
    }

    function handleRangeType(e: Event) {
        const type = (e.target as HTMLSelectElement).value as 'week' | 'month' | 'custom';
        rangeType = type;
        const now = new Date();
        let start = new Date(now);
        let end = new Date(now);
        if (type === 'week') {
            start.setDate(now.getDate() - now.getDay());
            end.setDate(start.getDate() + 6);
        } else if (type === 'month') {
            start = new Date(now.getFullYear(), now.getMonth(), 1);
            end = new Date(now.getFullYear(), now.getMonth() + 1, 0);
        }
        dispatch('rangechange', { type, start, end });
    }

    function handleCustomDate(field: 'start' | 'end', value: string) {
        const date = new Date(value);
        if (field === 'start') rangeStart = date;
        else rangeEnd = date;
        dispatch('rangechange', { type: 'custom', start: rangeStart, end: rangeEnd });
    }

    function toInputDate(d: Date): string {
        return d.toISOString().slice(0, 10);
    }
</script>
```

```svelte
<script lang="ts">
    import { onMount, onDestroy, createEventDispatcher } from 'svelte';
    import { tasks as taskApi } from '$lib/api';
    import { toast } from 'svelte-sonner';
    import { format } from 'date-fns';

    // READ PLAN 02 SUMMARY — use the import that matches installed package:
    // Option A (svelte-gantt original — D-01):
    //   import { SvelteGantt, SvelteGanttTable } from 'svelte-gantt';
    // Option B (@svar-ui/svelte-gantt Svelte 5 fallback):
    //   import { Gantt } from '@svar-ui/svelte-gantt';

    export let projects: any[] = [];
    export let viewMode: 'project' | 'member' = 'project';
    export let rangeStart: Date = new Date();
    export let rangeEnd: Date = new Date();

    const dispatch = createEventDispatcher();

    let ganttEl: HTMLElement;
    let ganttInstance: any;

    // Build rows + tasks from projects data based on viewMode
    function buildGanttData(projects: any[], mode: 'project' | 'member') {
        const rows: any[] = [];
        const ganttTasks: any[] = [];
        const now = new Date();

        if (mode === 'project') {
            for (const project of projects) {
                // Project header row
                rows.push({ id: `p-${project.id}`, label: project.name, enableDragging: false });

                // Milestone rows + their tasks
                for (const milestone of project.milestones) {
                    rows.push({ id: `m-${milestone.id}`, label: milestone.title, enableDragging: false });
                    for (const task of milestone.tasks) {
                        ganttTasks.push(buildTask(task, `m-${milestone.id}`, project.color));
                    }
                }

                // "No Milestone" catch-all row (D-04)
                if (project.unassigned_tasks.length > 0) {
                    rows.push({ id: `nm-${project.id}`, label: 'No Milestone', enableDragging: false });
                    for (const task of project.unassigned_tasks) {
                        ganttTasks.push(buildTask(task, `nm-${project.id}`, project.color));
                    }
                }
            }
        } else {
            // Member view: collect all tasks keyed by assignee
            const memberMap = new Map<string, { label: string; tasks: any[] }>();
            const unassignedTasks: any[] = [];

            for (const project of projects) {
                const allTasks = [
                    ...project.milestones.flatMap((m: any) => m.tasks),
                    ...project.unassigned_tasks
                ];
                for (const task of allTasks) {
                    if (task.assignee) {
                        const key = String(task.assignee.id);
                        if (!memberMap.has(key)) {
                            memberMap.set(key, { label: task.assignee.full_name, tasks: [] });
                        }
                        memberMap.get(key)!.tasks.push({ task, color: project.color });
                    } else {
                        unassignedTasks.push({ task, color: project.color });
                    }
                }
            }

            // Sort members alphabetically (Claude's Discretion)
            const sortedMembers = Array.from(memberMap.entries()).sort((a, b) => a[1].label.localeCompare(b[1].label));
            for (const [memberId, { label, tasks: memberTasks }] of sortedMembers) {
                rows.push({ id: `u-${memberId}`, label, enableDragging: false });
                for (const { task, color } of memberTasks) {
                    ganttTasks.push(buildTask(task, `u-${memberId}`, color));
                }
            }

            if (unassignedTasks.length > 0) {
                rows.push({ id: 'u-unassigned', label: 'Unassigned', enableDragging: false });
                for (const { task, color } of unassignedTasks) {
                    ganttTasks.push(buildTask(task, 'u-unassigned', color));
                }
            }
        }

        return { rows, tasks: ganttTasks };
    }

    function buildTask(task: any, rowId: string, color: string) {
        const now = new Date();
        const isUnscheduled = !task.due_date;
        const dueDate = task.due_date ? new Date(task.due_date) : now;
        const fromDate = task.due_date
            ? new Date(Math.min(new Date(task.created_at).getTime(), dueDate.getTime()))
            : now;

        const isOverdue = !isUnscheduled && dueDate < now && task.status !== 'done';

        return {
            id: task.id,
            resourceId: rowId,
            label: task.title,
            from: fromDate,
            to: dueDate,
            classes: [
                'gantt-task-bar',
                isUnscheduled ? 'task-unscheduled' : '',
                isOverdue ? 'task-overdue' : '',
                task.status === 'done' ? 'task-done' : ''
            ].filter(Boolean).join(' '),
            // Store extra data for click handler
            _taskData: task,
            _color: color,
        };
    }

    async function handleDrag(event: CustomEvent) {
        const { task } = event.detail;
        const newDueDate = task.to instanceof Date ? task.to : new Date(task.to);
        try {
            await taskApi.update(task.id, { due_date: newDueDate.toISOString() });
            toast.success(`Rescheduled to ${format(newDueDate, 'MMM d')}`);
            dispatch('reschedule', { taskId: task.id, newDueDate });
        } catch (err: any) {
            toast.error(`Failed to reschedule: ${err.message}`);
        }
    }

    // Reactive gantt data
    $: ganttData = buildGanttData(projects, viewMode);

    onMount(() => {
        // Note: Gantt initialization depends on which library was installed.
        // If using svelte-gantt (D-01 original):
        //   ganttInstance = new SvelteGantt({ target: ganttEl, props: { ... } });
        //   ganttInstance.$on('change', handleDrag);
        // The reactive $: ganttData will update, and the gantt needs refresh.
        // See TimelineGantt implementation notes in CONTEXT.md.
        //
        // If using @svar-ui/svelte-gantt (Svelte 5 fallback), mount via
        // the component template section using {#key} to re-render on data change.
    });

    onDestroy(() => {
        if (ganttInstance && typeof ganttInstance.$destroy === 'function') {
            ganttInstance.$destroy();
        }
    });
</script>
```

```svelte
<script lang="ts">
    import { onMount } from 'svelte';
    import { timeline, tasks as taskApi } from '$lib/api';
    import { toast } from 'svelte-sonner';
    import TimelineToolbar from '$lib/components/timeline/TimelineToolbar.svelte';
    import TimelineGantt from '$lib/components/timeline/TimelineGantt.svelte';
    import { format } from 'date-fns';
    import { X, Save } from 'lucide-svelte';

    type TaskStatus = 'todo' | 'in_progress' | 'review' | 'done' | 'blocked';
    type TaskPriority = 'low' | 'medium' | 'high' | 'critical';

    let projects: any[] = [];
    let loading = true;
    let error = '';

    // Toolbar state
    let viewMode: 'project' | 'member' = 'project';
    let rangeType: 'week' | 'month' | 'custom' = 'month';
    let rangeStart: Date = new Date(new Date().getFullYear(), new Date().getMonth(), 1);
    let rangeEnd: Date = new Date(new Date().getFullYear(), new Date().getMonth() + 1, 0);

    // Modal state (D-06)
    let selectedTask: any = null;
    let editForm = { title: '', status: '' as TaskStatus, priority: '' as TaskPriority, due_date: '', description: '' };
    let saving = false;

    async function loadTimeline() {
        loading = true;
        error = '';
        try {
            projects = await timeline.get();
            // D-08: fit to data — compute range from earliest/latest task dates
            fitToData();
        } catch (e: any) {
            error = e.message || 'Failed to load timeline';
        } finally {
            loading = false;
        }
    }

    function fitToData() {
        let earliest: Date | null = null;
        let latest: Date | null = null;
        const now = new Date();

        for (const project of projects) {
            const allTasks = [
                ...project.milestones.flatMap((m: any) => m.tasks),
                ...project.unassigned_tasks
            ];
            for (const task of allTasks) {
                const due = task.due_date ? new Date(task.due_date) : now;
                const start = new Date(task.created_at);
                if (!earliest || start < earliest) earliest = start;
                if (!latest || due > latest) latest = due;
            }
        }

        if (earliest && latest) {
            rangeStart = earliest;
            rangeEnd = latest;
            rangeType = 'custom';
        }
    }

    function handleViewChange(e: CustomEvent) {
        viewMode = e.detail;
    }

    function handleRangeChange(e: CustomEvent) {
        rangeType = e.detail.type;
        rangeStart = e.detail.start;
        rangeEnd = e.detail.end;
    }

    function handleTaskClick(e: CustomEvent) {
        const task = e.detail;
        selectedTask = task;
        editForm = {
            title: task.title,
            status: task.status,
            priority: task.priority,
            due_date: task.due_date ? task.due_date.slice(0, 10) : '',
            description: task.description || ''
        };
    }

    async function handleReschedule(e: CustomEvent) {
        // Drag already sent PATCH in TimelineGantt — just refresh
        await loadTimeline();
    }

    async function saveEdit() {
        if (!selectedTask) return;
        saving = true;
        try {
            const payload: any = {
                title: editForm.title,
                status: editForm.status,
                priority: editForm.priority,
            };
            if (editForm.due_date) payload.due_date = new Date(editForm.due_date).toISOString();
            if (editForm.description !== undefined) payload.description = editForm.description;

            await taskApi.update(selectedTask.id, payload);
            toast.success('Task updated');
            selectedTask = null;
            await loadTimeline();  // D-07: refresh after edit
        } catch (e: any) {
            toast.error(e.message || 'Failed to save');
        } finally {
            saving = false;
        }
    }

    onMount(loadTimeline);
</script>
```
