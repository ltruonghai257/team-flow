**Section 1 — my_tasks (all roles, D-02 to D-05):**

Query tasks assigned to the current user, excluding done tasks:
```python
now = datetime.now(timezone.utc).replace(tzinfo=None)
due_soon_cutoff = now + timedelta(hours=48)

task_result = await db.execute(
    select(Task)
    .outerjoin(CustomStatus, Task.custom_status_id == CustomStatus.id)
    .outerjoin(Project, Task.project_id == Project.id)
    .where(
        Task.assignee_id == current_user.id,
        Task.status != TaskStatus.done,
        (Task.custom_status_id.is_(None)) | (CustomStatus.is_done.is_(False)),
    )
    .options(selectinload(Task.custom_status))
)
tasks = task_result.scalars().all()
```

Build `DashboardTaskItem` list and sort in Python:
```python
task_items = []
for t in tasks:
    is_overdue = t.due_date is not None and t.due_date < now.date()
    is_due_soon = (
        not is_overdue
        and t.due_date is not None
        and now.date() <= t.due_date <= due_soon_cutoff.date()
    )
    project_name = t.project.name if t.project else None  # requires Project join above
    task_items.append(DashboardTaskItem(
        id=t.id,
        title=t.title,
        project_name=project_name,
        status=t.status,
        priority=t.priority,
        due_date=t.due_date,
        is_overdue=is_overdue,
        is_due_soon=is_due_soon,
    ))

# Sort: overdue first (due_date ASC), then upcoming (due_date ASC), then no due_date
overdue = sorted([x for x in task_items if x.is_overdue], key=lambda x: x.due_date)
upcoming = sorted([x for x in task_items if not x.is_overdue and x.due_date is not None], key=lambda x: x.due_date)
no_date = [x for x in task_items if x.due_date is None]
my_tasks = (overdue + upcoming + no_date)[:20]
```

Note: `t.project` is available because of `outerjoin(Project, ...)` but SQLAlchemy lazy-loading may not populate the relationship. Use `selectinload(Task.project)` instead of the outerjoin for project_name:
```python
# Replace the outerjoin(Project) with selectinload in options():
.options(selectinload(Task.custom_status), selectinload(Task.project))
# And simplify the where clause to not join Project:
.where(
    Task.assignee_id == current_user.id,
    Task.status != TaskStatus.done,
    (Task.custom_status_id.is_(None)) | (CustomStatus.is_done.is_(False)),
)
# Use outerjoin(CustomStatus...) only
```
Then: `project_name = t.project.name if t.project else None`
