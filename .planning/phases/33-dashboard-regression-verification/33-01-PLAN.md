import pytest
from fastapi.testclient import TestClient
from sqlalchemy.orm import Session

def test_get_dashboard_as_member(client: TestClient, db: Session, member_user):
    """Test dashboard endpoint returns member-only payload (no team_health or kpi_summary)"""
    response = client.get("/api/dashboard/", headers=member_user["headers"])
    assert response.status_code == 200
    data = response.json()
    assert "my_tasks" in data
    assert "recent_activity" in data
    assert data.get("team_health") is None
    assert data.get("kpi_summary") is None
